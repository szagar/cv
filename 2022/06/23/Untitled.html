def is_kang_word(word,kang) -> bool:
    word = word.upper()
    kang = kang.upper()
    if len(word) == len(kang): return False
    k_ptr = w_ptr = 0
    while True:
        if kang[k_ptr] == word[w_ptr]:
            k_ptr += 1
        w_ptr += 1
        if k_ptr == len(kang): 
            return True
        if w_ptr == len(word): break
    return False
